/* GAME STATE */


let difficulty

let players=1

let scores=[0,0]

let player=0

let question=0

let questions=[]


function selectPlayers(p){

players=p

document.getElementById("start").classList.add("hidden")

document.getElementById("difficulty").classList.remove("hidden")

}


function startGame(diff){


difficulty=diff


for(let i=0;i<24;i++){

questions.push(breeds[Math.floor(Math.random()*breeds.length)])

}


document.getElementById("difficulty").classList.add("hidden")

document.getElementById("quiz").classList.remove("hidden")


nextQuestion()


}


function nextQuestion(){


if(question>=24){

endGame()

return

}


let round=Math.floor(question/6)+1

document.getElementById("round").innerText="Round "+round+" / 4"


document.getElementById("turn").innerText=

players==2?"Player "+(player+1)+" turn":""



let q=questions[question]


let img="https://source.unsplash.com/600x400/?"+encodeURIComponent(q.breed+" dog")

document.getElementById("dogImage").src=img


let correct


if(difficulty=="easy"){

correct=q.group

document.getElementById("question").innerText="Which group is this dog?"

}


if(difficulty=="medium"){

correct=q.breed

document.getElementById("question").innerText="What breed is this dog?"

}


if(difficulty=="hard"){

correct=q.group+" — "+q.breed

document.getElementById("question").innerText="Which breed AND group?"

}


let options=[correct]


while(options.length<4){


let r=breeds[Math.floor(Math.random()*breeds.length)]


let option


if(difficulty=="easy") option=r.group

if(difficulty=="medium") option=r.breed

if(difficulty=="hard") option=r.group+" — "+r.breed


if(!options.includes(option)) options.push(option)


}


options.sort(()=>Math.random()-0.5)


let answers=document.getElementById("answers")

answers.innerHTML=""


options.forEach(o=>{


let b=document.createElement("button")

b.innerText=o

b.onclick=()=>select(o,correct)


answers.appendChild(b)


})


document.getElementById("score").innerText="Scores: "+scores.join(" / ")


}


function select(choice,correct){


if(choice==correct){

scores[player]++

}


question++


if(players==2){

player=(player+1)%2

}


nextQuestion()


}


function endGame(){


document.getElementById("quiz").classList.add("hidden")

document.getElementById("end").classList.remove("hidden")


document.getElementById("final").innerText="Final score: "+scores.join(" / ")


loadScores()


}


function saveScore(){


let name=document.getElementById("name").value||"Player"


let total=scores.reduce((a,b)=>a+b)


let board=JSON.parse(localStorage.getItem("dogquiz")||"[]")


board.push({name,score:total})


board.sort((a,b)=>b.score-a.score)


board=board.slice(0,10)


localStorage.setItem("dogquiz",JSON.stringify(board))


loadScores()


}


function loadScores(){


let board=JSON.parse(localStorage.getItem("dogquiz")||"[]")


let list=document.getElementById("leaderboard")


list.innerHTML=""


board.forEach(s=>{

let li=document.createElement("li")

li.innerText=s.name+" — "+s.score

list.appendChild(li)

})


}
